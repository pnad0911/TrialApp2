099    /**
100     * Create stringSQL
101     * @param strTable - table name
102     * @param strFileName - file name
103     * @return - String SQL
104     * @throws Exception
105     */
106    private static String stringSQL(String strTable, String strFileName) throws Exception {
107        INET net = new INET();
108        String strPage;
109        //check to see if the file exists on disk
110        Boolean boTest = net.fileExist(strFileName);
111        if (boTest == false) {
112            if (strFileName.equals("data/Grades.txt")) {
113                strPage = net.getURLRaw("http://faculty.sdmiramar.edu/jcouture/2014sp/cisc190/webct/manual/schema-grades.txt");
114                //strTest = net.getPREData(strPage);
115                net.saveToFile(strFileName, strPage);
116            } else if (strFileName.equals("data/Roster.txt")) {
117                //download the Winds Aloft data from the NWS and save it
118                strPage = net.getURLRaw("http://faculty.sdmiramar.edu/jcouture/2014sp/cisc190/webct/manual/schema-roster.txt");
119                //strTest = net.getPREData(strPage);
120                net.saveToFile(strFileName, strPage);
121            }
122        }
123        String str = "";
124        File myFile = new File(strFileName);
125        if (myFile.exists()) {
126            int intCount = 0;
127            Scanner inputFile = new Scanner(myFile);
128            StringBuilder strTGD = new StringBuilder("CREATE TABLE " + strTable + " (");
129            while (inputFile.hasNext()) {
130                String strLine;
131                strLine = inputFile.nextLine();
132                if (intCount > 9) {
133                    String[] tokens = strLine.split("\t");
134                    strTGD.append(tokens[0] + " ");
135                    if (strTable.equals(strTableGrades)) {
136                        aryGrades.add(tokens[0]);
137                    } else {
138                        aryRoster.add(tokens[0]);
139                    }
140                    strTGD.append(tokens[1].substring(3, tokens[1].length()).toUpperCase());
141                    if (tokens[2].trim().equals("NO") & intCount == 10) {
142                        strTGD.append(" NOT NULL");
143                        if (tokens[3].trim().equals("PRI")) {
144                            strTGD.append(" PRIMARY KEY, ");
145                        } else {
146                            strTGD.append(", ");
147                        }
148                    } else {
149                        strTGD.append(", ");
150                    }
151                }
152                intCount++;
153            }
154            strTGD.replace(strTGD.length() - 2, strTGD.length() - 1, ")");
155            str = strTGD.toString();
156        }
157
158        return str;
159    }

242    /**
243     * getQuizTotal(strStuID) that will return a "float" representing the total points of the student's quizzes
244     * @param strStuID - student ID
245     * @return - Total points of student's quiz
246     */
247    private static float getQuizTotal(String strStuID) {
248        DBUpdt db = new DBUpdt();
249        db.openConnection(strData);
250        //find stuid
251        db.query("SELECT * FROM " + strTableGrades + " WHERE stuid = '" + strStuID + "'");
252        int intCou;
253        float fl;
254        float tt = 0000;
255        while (db.moreRecords()) {
256            for (intCou = 21; intCou < 36; intCou++) {
257                String strS = db.getField(aryGrades.get(intCou));
258                //Divide that score by 100 and multiply it by 2
259                fl = (Float.valueOf(strS) / 100) * 2;
260                tt = tt + fl;
261            }
262        }
263        return tt;
264    }

266    /**
267     * getAsgnTotal(strStuID) that will return a "float"  representing the total points of the student's assignments
268     * @param strStuID - student ID
269     * @return -  total points of the student's assignments
270     */
271    private static float getAsgnTotal(String strStuID) {
272        int intCou;
273        int intCou1 = 0;
274        float fl;
275        float tt = 0000;
276        //Initialize an array of point values for each assignment using the JCouture.net course overview page as your guide
277        String[] ary = new String[15];
278        String strValueAsgn = "22233344455679";
279        int intC;
280        for (intC = 0; intC < strValueAsgn.length(); intC++) {
281            ary[intC] = String.valueOf(strValueAsgn.charAt(intC));
282        }
283        ary[14] = "11";
284        DBUpdt db = new DBUpdt();
285        db.openConnection(strData);
286        //Look up the student grades in the GRADES table
287        db.query("SELECT * FROM " + strTableGrades + " WHERE stuid = '" + strStuID + "'");
288        while (db.moreRecords()) {
289            for (intCou = 6; intCou < 21; intCou++) {
290                //Calculate the number points the student earned by taking the assignment score and dividing it by 100 then
291                // multiply it by the number of points for that assignment
292                String strS = db.getField(aryGrades.get(intCou));
293                fl = (Float.valueOf(ary[intCou1]) / 100) * (Float.valueOf(strS));
294                tt = tt + fl;
295                intCou1++;
296            }
297        }
298        return tt;
299    }

301    /**
302     * getLetterGrade(fltTotal) that accepts the total point value as a float and
303     * returns a single letter string representing the ordinal grade
304     * @param fltTotal - total points
305     * @return - letter grade
306     */
307    private static String getLetterGrade(float fltTotal) {
308        String strLetterGrade = "";
309        if (fltTotal < 60) {
310            strLetterGrade = "F";
311        } else if (fltTotal >= 60 & fltTotal < 70) {
312            strLetterGrade = "D";
313        } else if (fltTotal >= 70 & fltTotal < 80) {
314            strLetterGrade = "C";
315        } else if (fltTotal >= 80 & fltTotal < 90) {
316            strLetterGrade = "B";
317        } else if (fltTotal >= 90) {
318            strLetterGrade = "A";
319        }
320        return strLetterGrade;
321    }

323    /**
324     * Create Report
325     */
326    private static void CreateReport() {
327        DBUpdt db = new DBUpdt();
328        db.openConnection(strData);
329        db.query("SELECT roster.*, grades.* FROM " + strTableRoster + " INNER JOIN grades ON roster.stuid=grades.stuid ORDER BY lastname,firstname");
330        //write a report to a file called SEMESTER.TXT
331        File file = new File("data/SEMESTER.TXT");
332        PrintWriter output = null;
333        try {
334            // write a file to disk using the filename
335            output = new PrintWriter(file);
336        } catch (Exception egg) {
337            egg.printStackTrace();
338        }
339        while (db.moreRecords()) {
340            DecimalFormat format = new DecimalFormat("#0.00");
341            float flAsgn = getAsgnTotal(db.getField("stuid"));
342            float flQuiz = getQuizTotal(db.getField("stuid"));
343            output.println(db.getField("lastname").trim() + ";" + db.getField("firstname").trim()
344                    + ";" + db.getField("stuid").trim() + ";" + db.getField("email").trim() + ";" +
345                    String.valueOf(format.format(flQuiz)) + ";" + String.valueOf(format.format(flAsgn)) + ";" +
346                    String.valueOf(format.format((flAsgn + flQuiz))) + ";" + getLetterGrade(flAsgn + flQuiz));
347        }
348        db.close();
349        output.close();
350    }

428        public void actionPerformed(ActionEvent e) {
429            String selection13;
430            selection13 = (String) comboBox1.getSelectedItem();
431            String[] tokens = selection13.split(",");
432            if (blnChange(selection13) == true) {
433                DBUpdt db = new DBUpdt();
434                db.openConnection(strData);
435                db.query("SELECT * FROM " + strTableRoster + " WHERE lastname = '" + tokens[0] + "' AND firstname ='" + tokens[1] + "'");
436                while (db.moreRecords()) {
437                    String strID13 = db.getField("stuid");
438                    lblStuID.setText(strID13);
439                    DecimalFormat format = new DecimalFormat("#0.00");
440                    float flAsgn = getAsgnTotal(strID13);
441                    float flQuiz = getQuizTotal(strID13);

443                    lblQuiz.setText(String.valueOf(format.format(flQuiz)));
444                    lblTotal.setText(String.valueOf(format.format(flAsgn + flQuiz)));
445                    lblLetter.setText(getLetterGrade((flAsgn + flQuiz)));
